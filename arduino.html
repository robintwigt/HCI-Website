Arduino

Arduino is een open-source elektronica-platform. Met een Arduino kun je prototypen bouwen en interactieve objecten of omgevingen creëren. Arduino kan gebruikt worden zonder een computer, maar kan ook communiceren met een software die op een computer draait. Met de Arduino-microcontroller kan de input van sensoren ontvangen worden en kan er output naar actoren verzonden worden.

// the setup function runs once when you press reset or power the board
void setup() {
// initialize digital pin LED_BUILTIN as an output.
pinMode(13, OUTPUT);
pinMode(12, OUTPUT);
}

// the loop function runs over and over again forever
void loop() {
digitalWrite(13, HIGH); // turn the LED on (HIGH is the voltage level)
delay(0); // wait for a second
digitalWrite(12, LOW); // turn the LED on (HIGH is the voltage level)
delay(500); // wait for a second
digitalWrite(13, LOW); // turn the LED off by making the voltage LOW
delay(500); // wait for a second
digitalWrite(12, HIGH); // turn the LED off by making the voltage LOW
delay(0); // wait for a second
digitalWrite(13, HIGH); // turn the LED off by making the voltage LOW
delay(500); // wait for a second
digitalWrite(13, LOW); // turn the LED off by making the voltage LOW
delay(500); // wait for a second
}

Exercise 6C
Bij deze opdracht moest ik werken met een potentiometer. Als je eraan draait moet het ene lampje langzaam uit gaan en het andere lampje langzaam aan.

int greenLedPin = 10;
int yellowLedPin = 9;

float sensorValue1 = 0.0; // variable for sensor value
float sensorValue2 = 0.0; // variable for sensor value
int sensorPin = A0; // variable for sensor pin

void setup(){
pinMode(greenLedPin, OUTPUT);
pinMode(yellowLedPin, OUTPUT);
pinMode(sensorPin, INPUT);
Serial.begin(9600);
}

void loop() {
sensorValue1 = analogRead(sensorPin); // read the value/voltage on the sensor pin and
sensorValue1 = (sensorValue1/1023)*255; // store that value in the variable sensorValue
sensorValue1 = map(sensorValue1, 0, 255, 0, 255);
sensorValue2 = analogRead(sensorPin); // read the value/voltage on the sensor pin and
sensorValue2 = (sensorValue2/1023)*255; // store that value in the variable sensorValue
sensorValue2 = map(sensorValue2, 255, 0, 0, 255);
analogWrite(greenLedPin, sensorValue1);

Exercise 7
Bij deze opdracht moest ik gebruik maken van een licht sensor (LDR). Als er geen licht bij de sensor komt, moeten beide lampjes uit gaan. Als er wel licht bij de sensor komt, moeten beide lampjes weer aan gaan.

int greenLedPin = 10;
int yellowLedPin = 9;
int sensorValue1 = 9;
int sensorValue2 = 10;

float sensorValue = 0.0; // variable for sensor value
int sensorPin = A0; // variable for sensor pin

void setup() {
pinMode(greenLedPin, OUTPUT);
pinMode(yellowLedPin, OUTPUT);
pinMode(sensorPin, INPUT);
Serial.begin(9600);
}

void loop() {
sensorValue = analogRead(sensorPin); // read the value/voltage on the sensor pin and
sensorValue1 = map(sensorValue, 400, 700, 0, 255);
sensorValue2 = map(sensorValue, 400, 700, 0, 255);
sensorValue1 = constrain(sensorValue1, 0, 255); // limits range of sensor values to between 10 and 150
sensorValue2 = constrain(sensorValue2, 0, 255); // limits range of sensor values to between 10 and 150
analogWrite(greenLedPin, sensorValue1);
analogWrite(yellowLedPin, sensorValue2);

Serial.println(sensorValue1); // print out sensorValue to the Serial Monitor
delay(200); // delay for 0.2 seconds
}

Exercise 10F
Bij deze opdracht moest ik er voor zorgen dat de LED aan gaat wanneer er 1 button is ingedrukt, maar niet wanneer ze allebei ingedrukt zijn.

// constants won't change. They're used here to set pin numbers:
const int buttonPin1 = 2; // the number of the pushbutton pin
const int buttonPin2 = 3; // the number of the pushbutton pin
const int ledPin = 13; // the number of the LED pin

void setup() {
// initialize the LED pin as an output:
pinMode(ledPin, OUTPUT);
// initialize the pushbutton pin as an input:
pinMode(buttonPin1, INPUT);
// initialize the pushbutton pin as an input:
pinMode(buttonPin2, INPUT);
}

void loop() {
// check if the pushbutton is pressed. If it is, the buttonState is HIGH:
if ((digitalRead(buttonPin1) == HIGH) && (digitalRead(buttonPin2) == HIGH)) {
// turn LED on:
digitalWrite(ledPin, LOW);
} else if ((digitalRead(buttonPin1) == HIGH) || (digitalRead(buttonPin2) == HIGH)) {
// turn LED on:
digitalWrite(ledPin, HIGH);
} else {
// turn LED off:
digitalWrite(ledPin, LOW);
}
}

Exercise 11B
Bij deze opdracht moest ik gebruik maken van servomotor. De servo moest ik op een interessante manier laten bewegen.

#include < Servo.h>
Servo myServo; // Declare a servo object for us to control
int servoPin = 9; // Var for the pin the servo connects to
int pos = 0; // Var to keep track of the servo's position

void setup() {
myServo.attach(servoPin); // Tell the servo to what pin it's connected to
}

void loop() {
for(pos = 0; pos < 180; pos += 1){ // Loop, pos is added to (from 0 to 160)
myServo.write(pos); // Turn the servo to the position in pos
delay(15); // Wait 15ms
myServo.write(0); // Turn the servo to the position in pos
delay(500); // Wait 15ms
myServo.write(30); // Turn the servo to the position in pos
delay(1000); // Wait 15ms
myServo.write(0); // Turn the servo to the position in pos
delay(200); // Wait 15ms
myServo.write(120); // Turn the servo to the position in pos
delay(1000); // Wait 15ms
myServo.write(80); // Turn the servo to the position in pos
delay(200); // Wait 15ms
}
}

Exercise 12B
Bij deze opdracht moest ik weer gebruik maken van de servo. Als je de ene button indrukt, beweegt de wijzer naar links en als je de andere button indrukt, beweegt de wijzer naar rechts.

#include < Servo.h>
Servo myServo;
int servoPin = 9;
int pos = 0;
int buttonPinL = 3;
int buttonPinR = 2;

void setup() {
myServo.attach(servoPin);
pinMode(buttonPinL, INPUT);
pinMode(buttonPinR, INPUT);
}

void loop() {
if (digitalRead(buttonPinL) == HIGH && digitalRead(buttonPinR) == LOW){
pos++;
if (pos > 160) {
pos = 160;
}
delay(5);
}
if (digitalRead(buttonPinR) == HIGH && digitalRead(buttonPinL) == LOW){
pos--;
if (pos < 0) {
pos = 0;
}
delay(5);
}
if (pos >= 0 && pos <= 160) {
myServo.write(pos);
}
}
