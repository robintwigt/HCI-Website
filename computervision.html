Assignment CV1
Bij deze opdracht moest ik gaan experimenteren met de webcam. Ik moest ervoor zorgen dat er een cirkel om gezichten getekend wordt.

import gab.opencv.*;
import processing.video.*;
import java.awt.*;

Capture video;
OpenCV opencv;

void setup() {
size(640, 480);
video = new Capture(this, 640/2, 480/2);
opencv = new OpenCV(this, 640/2, 480/2);
opencv.loadCascade(OpenCV.CASCADE_FRONTALFACE);

video.start();
}

void draw() {
scale(2);
opencv.loadImage(video);

image(video, 0, 0 );

stroke(255, 255, 0);
noFill();
Rectangle[] faces = opencv.detect();
println(faces.length);

for (int i = 0; i < faces.length; i++) {
println(faces[i].x + "," + faces[i].y);
ellipseMode(CORNER);
ellipse(faces[i].x, faces[i].y, faces[i].width, faces[i].height);
}
}

void captureEvent(Capture c) {
c.read();
}

Bonus Assignment
Bij deze opdracht moest je een afbeelding ergens plaatsen op een gezicht.

import gab.opencv.*;
import processing.video.*;
import java.awt.*;

Capture video;
OpenCV opencv;
PImage img;

void setup() {

image(video, 0, 0 );

Assignment AR
Bij deze opdracht moest je gebruik maken van papier met daarop een afbeelding. Als je dit papier dan voor de webcam houdt, dan krijg je iets te zien op de webcam.

/**
* NyARToolkit for proce55ing/3.0.5
* (c)2008-2017 nyatla
* airmail(at)ebony.plala.or.jp
*
* 最も短いARToolKitのコードです。
* Hiroマーカの上に立方体を表示します。
* 全ての設定ファイルとマーカファイルはスケッチディレクトリのlibraries/nyar4psg/dataにあります。
*
* This sketch is shortest sample.
* The sketch shows cube on the marker of "patt.hiro".
* Any pattern and configuration files are found in libraries/nyar4psg/data inside your sketchbook folder.
*/

import processing.video.*;
import jp.nyatla.nyar4psg.*;

Capture cam;
MultiMarker nya;

void setup() {
size(640,480,P3D);
colorMode(RGB, 100);
println(MultiMarker.VERSION);
cam=new Capture(this,640,480);
nya=new MultiMarker(this,width,height,"../../data/camera_para.dat",NyAR4PsgConfig.CONFIG_PSG);
nya.addARMarker("../../data/patt.hiro",80);
cam.start();
}

void draw()
{
if (cam.available() !=true) {
return;
}
cam.read();
nya.detect(cam);
background(0);
nya.drawBackground(cam);//frustumを考慮した背景描画
if((!nya.isExist(0))){
return;
}
nya.beginTransform(0);
fill(0,0,255);
translate(0,0,20);
box(40);
nya.endTransform();
}
